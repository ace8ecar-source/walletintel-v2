Dead simple integration

One HTTP request. No SDK, no API key, no auth.

cURL
Python
JavaScript
# Get full PnL analysis for any wallet curl "https://api.walletintel.dev/wallet/YOUR_WALLET/pnl" # Quick summary (no token breakdown) curl "https://api.walletintel.dev/wallet/YOUR_WALLET/summary" # Top wallets leaderboard curl "https://api.walletintel.dev/leaderboard?sort=score"
import requests wallet = "YOUR_WALLET_ADDRESS" resp = requests.get(f"https://api.walletintel.dev/wallet/{wallet}/pnl") data = resp.json() print(f"PnL: {data['total_realized_pnl_sol']} SOL") print(f"Win Rate: {data['win_rate']}%") print(f"Strategy: {data['strategy']}") print(f"Score: {data['score']}/100")
const wallet = "YOUR_WALLET_ADDRESS"; const res = await fetch( `https://api.walletintel.dev/wallet/${wallet}/pnl` ); const data = await res.json(); console.log(`PnL: ${data.total_realized_pnl_sol} SOL`); console.log(`Score: ${data.score}/100`);
